Exemplo em React:

jsx
import React, { useState } from 'react';

function App() {
  const [copied, setCopied] = useState(false);
  const textoParaCopiar = "Este é o texto que você pode copiar!";

  const copiarTexto = async () => {
    try {
      await navigator.clipboard.writeText(textoParaCopiar);
      setCopied(true); // Define como "copiado" para mostrar um feedback
      setTimeout(() => setCopied(false), 2000); // Reseta após 2 segundos
    } catch (error) {
      alert('Erro ao copiar o texto: ' + error);
    }
  };

  return (
    <div>
      <div id="textoParaCopiar">{textoParaCopiar}</div>
      <button onClick={copiarTexto}>
        {copied ? 'Texto Copiado!' : 'Copiar Texto'}
      </button>
    </div>
  );
}

export default App;

1. Adicione scripts externos nopublic/index.html

Se o script não interagir diretamente com seus componentes React, você poderá adicioná-lo diretamente no arquivo public/index.html.

HTML->
<!-- public/index.html -->
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="utf-8" />
    <title>React App</title>
    <script src="https://example.com/external-script.js"></script> <!-- Adicione aqui -->
  </head>
  <body>
    <div id="root"></div>
  </body>
</html>

2. Carregar um script dentro de um componente React usandouseEffect

Se você precisar carregar e executar um script específico de forma dinâmica, poderá fazê-lo no React usando o hook useEffect.

jsx
import React, { useEffect } from "react";

function App() {
  useEffect(() => {
    // Cria a tag <script>
    const script = document.createElement("script");
    script.src = "https://example.com/external-script.js"; // URL do script
    script.async = true; // Torna o script assíncrono (opcional)
    document.body.appendChild(script); // Adiciona o script ao body

    // Limpeza: Remove o script quando o componente desmontar
    return () => {
      document.body.removeChild(script);
    };
  }, []); // [] garante que o script só seja carregado uma vez

  return (
    <div>
      <h1>React com Script Externo</h1>
    </div>
  );
}

export default App;
